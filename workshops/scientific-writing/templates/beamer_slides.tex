\section{Introduction}
% ==============================================================================

\begin{frame}{Why \LaTeX{}?}
    \begin{itemize}
        \item \textbf{Publication-quality typesetting} — especially for
              mathematics.
        \item \textbf{Structured documents} — automatic numbering, cross-%
              references, and bibliographies.
        \item \textbf{Reproducibility} — plain-text source under version
              control.
        \item \textbf{Wide adoption} — required by most journals in
              mathematics, physics, and computer science.
    \end{itemize}

    \vspace{1em}
    \begin{block}{Key Idea}
        You describe \emph{what} to typeset; \LaTeX{} decides \emph{how}.
    \end{block}
\end{frame}

% ==============================================================================
\section{Equations}
% ==============================================================================

\begin{frame}{Writing Mathematics}
    Inline math: $E = mc^{2}$.

    \medskip
    Display equations with automatic numbering:
    \begin{equation}
        \int_{0}^{\infty} e^{-x^{2}} \, dx = \frac{\sqrt{\pi}}{2}
    \end{equation}

    \medskip
    Multi-line alignment:
    \begin{align}
        (a + b)^{2} &= a^{2} + 2ab + b^{2} \\
        (a - b)^{2} &= a^{2} - 2ab + b^{2}
    \end{align}
\end{frame}
