\chapter*{Preface}
\addcontentsline{toc}{chapter}{Preface}

Time-indexed data are ubiquitous: stock prices recorded every minute, daily temperatures, quarterly GDP, EEG signals, network traffic. \textbf{Time series analysis} provides the mathematical framework for modeling the \emph{temporal dependence} inherent in such data, and exploiting this structure for \emph{forecasting}, \emph{anomaly detection}, and \emph{understanding} the underlying mechanisms.

\medskip
This course, aimed at Master students in statistics, economics, and data science, covers classical theory (ARMA processes, spectral analysis, Kalman filter) and modern developments (GARCH, cointegration, VAR models). Each concept is illustrated with real data and accompanied by Python implementations.

\medskip
\textbf{Prerequisites.} Probability theory (convergences, conditional expectation), mathematical statistics (estimation, testing), real analysis, linear algebra.

\medskip

\chapter{Formula Sheet}
\label{app:formulas}

\section{ARMA Processes}
\begin{align}
\text{AR}(p) &: X_t = \phi_1 X_{t-1}+\cdots+\phi_p X_{t-p}+\eps_t \\
\text{MA}(q) &: X_t = \eps_t+\theta_1\eps_{t-1}+\cdots+\theta_q\eps_{t-q} \\
\text{ARMA}(p,q) &: \phi(B)X_t = \theta(B)\eps_t \\
\text{ARIMA}(p,d,q) &: \phi(B)(1-B)^d X_t = \theta(B)\eps_t
\end{align}

\section{Autocovariance and Autocorrelation}
\begin{align}
\gamma(h) &= \Cov(X_t,X_{t+h}) \\
\rho(h) &= \gamma(h)/\gamma(0) \\
\hat{\gamma}(h) &= \frac{1}{n}\sum_{t=1}^{n-h}(X_t-\bar{X})(X_{t+h}-\bar{X})
\end{align}

\section{Information Criteria}
\begin{align}
\AIC &= -2\ln L + 2k \\
\BIC &= -2\ln L + k\ln n
\end{align}

\section{GARCH}
\begin{align}
\text{GARCH}(1,1) &: \sigma_t^2 = \omega + \alpha\eps_{t-1}^2 + \beta\sigma_{t-1}^2
\end{align}

\section{Kalman Filter}
\begin{align}
\text{State} &: \bm{x}_{t+1} = F\bm{x}_t + \bm{w}_t \\
\text{Observation} &: \bm{y}_t = H\bm{x}_t + \bm{v}_t
\end{align}

\section{VAR}
\begin{align}
\text{VAR}(p) &: \bm{Y}_t = \bm{c} + A_1\bm{Y}_{t-1}+\cdots+A_p\bm{Y}_{t-p}+\bm{u}_t
\end{align}
