\chapter*{Pr\'eface}
\addcontentsline{toc}{chapter}{Pr\'eface}

Les donn\'ees index\'ees par le temps sont omnipr\'esentes : cours boursiers minute par minute, temp\'eratures journali\`eres, PIB trimestriel, signaux EEG, trafic r\'eseau. L'\textbf{analyse des s\'eries temporelles} fournit le cadre math\'ematique pour mod\'eliser la \emph{d\'ependance temporelle} inh\'erente \`a ces donn\'ees, et exploiter cette structure pour la \emph{pr\'evision}, la \emph{d\'etection d'anomalies} et la \emph{compr\'ehension} des m\'ecanismes sous-jacents.

\medskip
Ce cours, destin\'e aux \'etudiants de Master en statistique, \'economie et science des donn\'ees, couvre la th\'eorie classique (processus ARMA, analyse spectrale, filtre de Kalman) et les d\'eveloppements modernes (GARCH, coint\'egration, mod\`eles VAR). Chaque concept est illustr\'e par des donn\'ees r\'eelles et accompagn\'e d'impl\'ementations en Python.

\medskip
\textbf{Pr\'erequis.} Probabilit\'es (convergences, esp\'erance conditionnelle), statistique math\'ematique (estimation, tests), analyse r\'eelle, alg\`ebre lin\'eaire.

\medskip

\chapter{Formulaire}
\label{app:formulaire}

\section{Processus ARMA}
\begin{align}
\text{AR}(p) &: X_t = \phi_1 X_{t-1}+\cdots+\phi_p X_{t-p}+\eps_t \\
\text{MA}(q) &: X_t = \eps_t+\theta_1\eps_{t-1}+\cdots+\theta_q\eps_{t-q} \\
\text{ARMA}(p,q) &: \phi(B)X_t = \theta(B)\eps_t \\
\text{ARIMA}(p,d,q) &: \phi(B)(1-B)^d X_t = \theta(B)\eps_t
\end{align}

\section{Autocovariance et autocorr\'elation}
\begin{align}
\gamma(h) &= \Cov(X_t,X_{t+h}) \\
\rho(h) &= \gamma(h)/\gamma(0) \\
\hat{\gamma}(h) &= \frac{1}{n}\sum_{t=1}^{n-h}(X_t-\bar{X})(X_{t+h}-\bar{X})
\end{align}

\section{Crit\`eres d'information}
\begin{align}
\AIC &= -2\ln L + 2k \\
\BIC &= -2\ln L + k\ln n
\end{align}

\section{GARCH}
\begin{align}
\text{GARCH}(1,1) &: \sigma_t^2 = \omega + \alpha\eps_{t-1}^2 + \beta\sigma_{t-1}^2
\end{align}

\section{Filtre de Kalman}
\begin{align}
\text{\'Etat} &: \bm{x}_{t+1} = F\bm{x}_t + \bm{w}_t \\
\text{Observation} &: \bm{y}_t = H\bm{x}_t + \bm{v}_t
\end{align}

\section{VAR}
\begin{align}
\text{VAR}(p) &: \bm{Y}_t = \bm{c} + A_1\bm{Y}_{t-1}+\cdots+A_p\bm{Y}_{t-p}+\bm{u}_t
\end{align}
