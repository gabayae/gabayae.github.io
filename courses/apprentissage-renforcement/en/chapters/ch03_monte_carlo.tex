% =============================================================================
\chapter{Monte Carlo and TD Methods}
\label{ch:monte_carlo}
% =============================================================================

Dynamic programming assumes perfect knowledge of the environment: transition probabilities, rewards, everything is given. But in the real world, a robot learning to walk does not know the equations of physics---it only has its \emph{experience}, the trajectories it has actually followed. How can one learn from experience alone?

Two fundamental answers emerge. \emph{Monte Carlo} methods, heirs to the work of Stanislaw Ulam and John von Neumann at the Los Alamos laboratory in the 1940s, estimate values by averaging the returns observed over complete episodes. \emph{Temporal difference} (TD) methods, introduced by Richard Sutton in 1988, do something bolder: they update estimates at every time step, using their own predictions as targets---a mechanism called \emph{bootstrapping}. This intellectually dizzying idea (learning from what you do not yet know) is the keystone of modern reinforcement learning.

\begin{intuition}
Monte Carlo and Temporal Difference (TD) methods learn value functions directly
from experience without knowing the transition model. Monte Carlo waits until
the end of an episode to update; TD updates at every time step using a
bootstrap estimate.
\end{intuition}

% -----------------------------------------------------------------------------
\section{Model-Free Learning}
% -----------------------------------------------------------------------------

Unlike dynamic programming, which requires complete knowledge of the model
$(P, R)$, model-free methods learn from episodes generated by direct interaction
with the environment. An episode is a sequence:
\[
    S_0, A_0, R_1, S_1, A_1, R_2, \ldots, S_{T-1}, A_{T-1}, R_T, S_T
\]
where $S_T$ is a terminal state.

% -----------------------------------------------------------------------------
\section{Monte Carlo Methods}
% -----------------------------------------------------------------------------

\subsection{MC Value Estimation}

\begin{definition}[First-Visit MC Evaluation]
The \textbf{first-visit} MC estimate of $V^\pi(s)$ is the average of returns
$G_t$ observed at the first occurrence of $s$ in each episode:
\[
    V^\pi(s) \approx \frac{1}{N(s)} \sum_{i=1}^{N(s)} G_t^{(i)}
\]
where $N(s)$ is the number of episodes in which $s$ was visited.
\end{definition}

\begin{definition}[Every-Visit MC Evaluation]
The \textbf{every-visit} MC estimate uses all occurrences of $s$ in all
episodes, not just the first.
\end{definition}

\begin{theorem}[First-Visit MC Convergence]
The first-visit MC estimate converges to $V^\pi(s)$ as $N(s) \to \infty$,
by the strong law of large numbers, since the returns $G_t^{(i)}$ are i.i.d.\
random variables with expectation $V^\pi(s)$.
\end{theorem}

\begin{algorithme}[title=\textbf{First-Visit MC Evaluation}]
\begin{enumerate}
    \item Initialize $V(s) \leftarrow 0$, $N(s) \leftarrow 0$ for all $s$
    \item For each episode:
    \begin{enumerate}
        \item Generate episode following $\pi$: $S_0, A_0, R_1, \ldots, S_T$
        \item $G \leftarrow 0$
        \item For $t = T-1, T-2, \ldots, 0$:
        \begin{enumerate}
            \item $G \leftarrow \gamma G + R_{t+1}$
            \item If $S_t$ does not appear in $S_0, \ldots, S_{t-1}$:
            \begin{enumerate}
                \item $N(S_t) \leftarrow N(S_t) + 1$
                \item $V(S_t) \leftarrow V(S_t) + \frac{1}{N(S_t)}(G - V(S_t))$
            \end{enumerate}
        \end{enumerate}
    \end{enumerate}
\end{enumerate}
\end{algorithme}

\subsection{MC Estimation of the Q-Function}

For model-free policy improvement, it is necessary to estimate $Q^\pi(s,a)$
rather than $V^\pi(s)$:
\[
    Q^\pi(s,a) \approx \frac{1}{N(s,a)} \sum_{i=1}^{N(s,a)} G_t^{(i)}.
\]

\begin{warning}[title=\textbf{Exploration Problem}]
If $\pi$ is deterministic, some state-action pairs $(s,a)$ will never be
visited. Solutions: use \textbf{exploring starts} or an $\epsilon$-greedy policy.
\end{warning}

\subsection{MC with $\epsilon$-Greedy Policy}

\begin{definition}[$\epsilon$-Greedy Policy]
The $\epsilon$-greedy policy derived from $Q$ is:
\[
    \pi_\epsilon(a \mid s) = \begin{cases}
        1 - \epsilon + \frac{\epsilon}{|\mathcal{A}|} & \text{if } a = \argmax_{a'} Q(s,a') \\
        \frac{\epsilon}{|\mathcal{A}|} & \text{otherwise}
    \end{cases}
\]
\end{definition}

\begin{theorem}[$\epsilon$-Greedy Improvement]
The $\epsilon$-greedy policy $\pi'$ with respect to $Q^{\pi_\epsilon}$
satisfies $V^{\pi'}(s) \geq V^{\pi_\epsilon}(s)$ for all $s$.
\end{theorem}

\section{Off-Policy MC with Importance Sampling}
% -----------------------------------------------------------------------------

\begin{definition}[Importance Sampling Ratio]
Let $b$ be the behavior policy and $\pi$ the target policy. The
\textbf{importance sampling ratio} for a trajectory from $t$ to $T-1$ is:
\[
    \rho_{t:T-1} = \prod_{k=t}^{T-1} \frac{\pi(A_k \mid S_k)}{b(A_k \mid S_k)}.
\]
\end{definition}

\begin{keyformulas}[title=\textbf{Off-Policy MC Estimator}]
The off-policy estimate of $V^\pi$ is:
\[
    V^\pi(s) \approx \frac{\sum_{i} \rho_{t_i:T_i-1} G_{t_i}^{(i)}}
    {\sum_{i} \rho_{t_i:T_i-1}}
    \quad \text{(weighted importance sampling)}.
\]
\end{keyformulas}

\begin{proposition}[Bias and Variance]
The ordinary estimator $\hat{V} = \frac{1}{n}\sum_i \rho_i G_i$ is unbiased
but has potentially infinite variance. The weighted estimator
$\hat{V}_w = \frac{\sum_i \rho_i G_i}{\sum_i \rho_i}$ is biased but has
bounded variance. The bias vanishes as $n \to \infty$.
\end{proposition}

% -----------------------------------------------------------------------------
\section{Temporal Differences --- TD(0)}
% -----------------------------------------------------------------------------

\begin{definition}[TD(0) Update]
The TD(0) method updates $V$ at every time step using the \textbf{TD target}:
\[
    V(S_t) \leftarrow V(S_t) + \alpha \big[ \underbrace{R_{t+1} + \gamma V(S_{t+1})}_{\text{TD target}} - V(S_t) \big].
\]
The term $\delta_t = R_{t+1} + \gamma V(S_{t+1}) - V(S_t)$ is called the
\textbf{TD error}.
\end{definition}

\begin{keyformulas}[title=\textbf{Temporal Difference Error}]
\[
    \delta_t = R_{t+1} + \gamma V(S_{t+1}) - V(S_t)
\]
The TD error is an unbiased estimator of the advantage $A^\pi(S_t, A_t)$ in
expectation.
\end{keyformulas}

\begin{theorem}[Convergence of TD(0)]
Under the Robbins-Monro conditions on the learning rate:
\[
    \sum_{t=0}^{\infty} \alpha_t = \infty, \qquad
    \sum_{t=0}^{\infty} \alpha_t^2 < \infty,
\]
the TD(0) algorithm converges almost surely to $V^\pi$.
\end{theorem}

\begin{algorithme}[title=\textbf{TD(0) for Policy Evaluation}]
\begin{enumerate}
    \item Initialize $V(s)$ arbitrarily for all $s$, fix $\alpha$
    \item For each episode:
    \begin{enumerate}
        \item Initialize $S_0$
        \item For each step $t = 0, 1, 2, \ldots$ until termination:
        \begin{enumerate}
            \item $A_t \sim \pi(\cdot | S_t)$
            \item Observe $R_{t+1}, S_{t+1}$
            \item $V(S_t) \leftarrow V(S_t) + \alpha[R_{t+1} + \gamma V(S_{t+1}) - V(S_t)]$
        \end{enumerate}
    \end{enumerate}
\end{enumerate}
\end{algorithme}

% -----------------------------------------------------------------------------
\section{Comparison: MC vs TD}
% -----------------------------------------------------------------------------

\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{lcc}
    \toprule
    & \textbf{Monte Carlo} & \textbf{TD(0)} \\
    \midrule
    Requires complete episodes & Yes & No \\
    Bootstrapping & No & Yes \\
    Bias & None & Biased (diminishes) \\
    Variance & High & Low \\
    Convergence & $1/\sqrt{N}$ & Faster in practice \\
    Sensitivity to $\alpha$ & Low & High \\
    \bottomrule
\end{tabular}
\end{center}

\begin{remark}
The \textbf{bias-variance tradeoff} is central: MC has zero bias but high
variance (since the return $G_t$ integrates all future rewards). TD has bias
due to bootstrapping ($V(S_{t+1})$ is an estimate) but lower variance.
\end{remark}

\section{TD($\lambda$) and Eligibility Traces}
% -----------------------------------------------------------------------------

\begin{definition}[$\lambda$-Return]
The \textbf{$\lambda$-return} is an exponentially weighted average of $n$-step
returns:
\[
    G_t^\lambda = (1-\lambda) \sum_{n=1}^{\infty} \lambda^{n-1} G_t^{(n)}
\]
where $G_t^{(n)} = \sum_{k=0}^{n-1} \gamma^k R_{t+k+1} + \gamma^n V(S_{t+n})$
is the $n$-step return.
\end{definition}

\begin{definition}[Eligibility Trace]
The \textbf{eligibility trace} for state $s$ at time $t$ is:
\[
    e_t(s) = \begin{cases}
        \gamma \lambda \, e_{t-1}(s) + 1 & \text{if } S_t = s \\
        \gamma \lambda \, e_{t-1}(s) & \text{otherwise}
    \end{cases}
\]
with $e_0(s) = \mathbb{1}_{S_0 = s}$.
\end{definition}

\begin{algorithme}[title=\textbf{TD($\lambda$) with Eligibility Traces}]
\begin{enumerate}
    \item Initialize $V(s)$ for all $s$
    \item For each episode:
    \begin{enumerate}
        \item $e(s) \leftarrow 0$ for all $s$
        \item For each step $t$:
        \begin{enumerate}
            \item Observe $R_{t+1}, S_{t+1}$
            \item $\delta_t \leftarrow R_{t+1} + \gamma V(S_{t+1}) - V(S_t)$
            \item $e(S_t) \leftarrow e(S_t) + 1$
            \item For all $s$: $V(s) \leftarrow V(s) + \alpha \delta_t e(s)$
            \item For all $s$: $e(s) \leftarrow \gamma \lambda \, e(s)$
        \end{enumerate}
    \end{enumerate}
\end{enumerate}
\end{algorithme}

\begin{remark}
$\lambda = 0$ yields TD(0); $\lambda = 1$ yields (essentially) MC. Intermediate
values $\lambda \in (0,1)$ offer an optimal bias-variance tradeoff in practice.
\end{remark}

\section{Python Implementation}
% -----------------------------------------------------------------------------

\begin{codeblock}[title=\textbf{First-Visit Monte Carlo Evaluation}]
\begin{minted}{python}
import numpy as np
import gymnasium as gym

def mc_first_visit(env, pi, n_episodes=10000, gamma=0.99):
    """First-visit MC evaluation of V^pi."""
    nS = env.observation_space.n
    V = np.zeros(nS)
    N = np.zeros(nS)

    for _ in range(n_episodes):
        episode = []
        s, _ = env.reset()
        done = False
        while not done:
            a = pi(s)
            s_next, r, terminated, truncated, _ = env.step(a)
            episode.append((s, a, r))
            s = s_next
            done = terminated or truncated

        G = 0.0
        visited = set()
        for s, a, r in reversed(episode):
            G = gamma * G + r
            if s not in visited:
                visited.add(s)
                N[s] += 1
                V[s] += (G - V[s]) / N[s]
    return V
\end{minted}
\end{codeblock}

\begin{codeblock}[title=\textbf{TD(0) for Policy Evaluation}]
\begin{minted}{python}
def td0_evaluation(env, pi, n_episodes=10000, alpha=0.01, gamma=0.99):
    """TD(0) evaluation of V^pi."""
    nS = env.observation_space.n
    V = np.zeros(nS)

    for _ in range(n_episodes):
        s, _ = env.reset()
        done = False
        while not done:
            a = pi(s)
            s_next, r, terminated, truncated, _ = env.step(a)
            done = terminated or truncated
            V_next = 0.0 if terminated else V[s_next]
            V[s] += alpha * (r + gamma * V_next - V[s])
            s = s_next
    return V
\end{minted}
\end{codeblock}

\begin{codeblock}[title=\textbf{TD($\lambda$) with Eligibility Traces}]
\begin{minted}{python}
def td_lambda(env, pi, n_episodes=10000, alpha=0.01,
              gamma=0.99, lam=0.8):
    """TD(lambda) with eligibility traces."""
    nS = env.observation_space.n
    V = np.zeros(nS)

    for _ in range(n_episodes):
        e = np.zeros(nS)
        s, _ = env.reset()
        done = False
        while not done:
            a = pi(s)
            s_next, r, terminated, truncated, _ = env.step(a)
            done = terminated or truncated
            V_next = 0.0 if terminated else V[s_next]
            delta = r + gamma * V_next - V[s]
            e[s] += 1.0  # accumulating trace
            V += alpha * delta * e
            e *= gamma * lam
            s = s_next
    return V
\end{minted}
\end{codeblock}

\section{Exercises}
% -----------------------------------------------------------------------------

\begin{exercise}[MC vs TD Variance]
Generate 1000 episodes in a 5-state MDP with a fixed policy. Estimate $V^\pi$
by first-visit MC and TD(0). Plot the error curves $\norm{V_n - V^\pi}_2$ as
a function of episode count $n$.
\end{exercise}

\begin{exercise}[$n$-Step Return]
Implement the $n$-step return $G_t^{(n)}$ and compare performance for
$n \in \{1, 2, 4, 8, \infty\}$ on the \texttt{FrozenLake-v1} environment.
\end{exercise}

\begin{exercise}[TD Convergence Proof]
Prove that the TD error $\delta_t$ satisfies $\E_\pi[\delta_t | S_t = s] = 0$
when $V = V^\pi$. What does this imply about the fixed point of TD(0)?
\end{exercise}

\begin{exercise}[Importance Sampling]
Implement off-policy MC estimation with weighted importance sampling to
estimate $V^\pi$ from trajectories generated by a uniform policy $b$.
Compare with the ordinary estimator.
\end{exercise}
