% ============================================================
\chapter{It\^o Calculus and Stochastic Differential Equations}
\label{ch:ito_sde}
% ============================================================

\begin{center}
\textit{``Stochastic calculus is the art of giving rigorous meaning
to integrals that classical analysis refuses.''}
\end{center}

\bigskip

This chapter develops the theory of the It\^o integral, establishes
It\^o's formula (the stochastic chain rule), and studies existence
and uniqueness of solutions to stochastic differential equations (SDEs).
We treat in detail geometric Brownian motion and the Ornstein--Uhlenbeck
process.

% ------------------------------------------------------------
\section{Construction of the It\^o Integral}
\label{sec:ito_integral}
% ------------------------------------------------------------

\begin{definition}[It\^o Integral for Simple Processes]
\index{It\^o integral}
Let $(B_t)_{t \geq 0}$ be a standard Brownian motion on
$(\Omega, \mathcal{F}, (\mathcal{F}_t), \PP)$. For a simple
(step) process $H_t = \sum_{i=0}^{n-1} h_i\,\ind_{(t_i, t_{i+1}]}(t)$
where each $h_i$ is $\mathcal{F}_{t_i}$-measurable and bounded, define:
\[
  \int_0^T H_s\,dB_s = \sum_{i=0}^{n-1} h_i\,(B_{t_{i+1}} - B_{t_i}).
\]
\end{definition}

\begin{theorem}[It\^o Isometry]
\label{thm:isometry}
For any adapted simple process $H$:
\[
  \E\left[\left(\int_0^T H_s\,dB_s\right)^2\right]
  = \E\left[\int_0^T H_s^2\,ds\right].
\]
\end{theorem}

\begin{proof}
Expand the square:
\[
  \E\left[\left(\sum_i h_i\,\Delta B_i\right)^2\right]
  = \sum_i \E[h_i^2]\,\E[(\Delta B_i)^2]
    + 2\sum_{i < j} \E[h_i\,\Delta B_i\,h_j\,\Delta B_j].
\]
The first sum gives $\sum_i \E[h_i^2]\,(t_{i+1} - t_i)$. For $i < j$,
$h_j\,\Delta B_j$ is independent of $\mathcal{F}_{t_j}$ and
$\E[\Delta B_j] = 0$, so the cross terms vanish. Hence
$\E[(\int_0^T H\,dB)^2] = \E[\int_0^T H_s^2\,ds]$.
\end{proof}

\begin{intuition}
The It\^o isometry is the stochastic analogue of Parseval's identity
in Fourier analysis. It allows extending the It\^o integral by
continuity to all adapted square-integrable processes, using
completeness of $L^2$.
\end{intuition}

\begin{definition}[Extension to $L^2$]
\index{space $\mathcal{H}^2$}
Let $\mathcal{H}^2_T$ denote the space of adapted processes
$(H_t)_{0 \leq t \leq T}$ with $\E[\int_0^T H_s^2\,ds] < \infty$.
The It\^o integral extends uniquely to $\mathcal{H}^2_T$ by density of
simple processes and the It\^o isometry.
\end{definition}

\begin{proposition}[Fundamental Properties]
\label{prop:ito_properties}
The It\^o integral $M_t = \int_0^t H_s\,dB_s$ satisfies:
\begin{enumerate}
  \item \textbf{Martingale:} $(M_t)$ is a square-integrable martingale.
  \item \textbf{Linearity:} $\int_0^t (\alpha H_s + \beta K_s)\,dB_s
    = \alpha\int_0^t H_s\,dB_s + \beta\int_0^t K_s\,dB_s$.
  \item \textbf{Continuous paths:} $t \mapsto M_t$ is a.s.\ continuous.
\end{enumerate}
\end{proposition}

\begin{attention}
The It\^o integral is \emph{not} a Riemann--Stieltjes integral because
Brownian paths have infinite variation. Evaluating the integrand at
the \emph{left} endpoint of each subinterval is crucial: choosing the
midpoint would give the Stratonovich integral\index{Stratonovich integral},
which obeys different calculus rules.
\end{attention}

\section{It\^o's Formula}
\label{sec:ito_formula}
% ------------------------------------------------------------

\begin{theorem}[It\^o's Formula, One Dimension]
\label{thm:ito_formula}
\index{It\^o's formula}
Let $X_t = X_0 + \int_0^t b_s\,ds + \int_0^t \sigma_s\,dB_s$ be an
It\^o process and $f \in C^2(\R)$. Then:
\[
  f(X_t) = f(X_0) + \int_0^t f'(X_s)\,b_s\,ds
    + \int_0^t f'(X_s)\,\sigma_s\,dB_s
    + \frac{1}{2}\int_0^t f''(X_s)\,\sigma_s^2\,ds.
\]
\end{theorem}

\begin{proof}[Proof sketch]
Write the Taylor expansion to second order:
$f(X_{t+dt}) - f(X_t) \approx f'(X_t)\,dX_t + \frac{1}{2}f''(X_t)\,(dX_t)^2$.
The key rule is $(dB_t)^2 = dt$ (in mean square), while $dt\,dB_t = 0$
and $(dt)^2 = 0$. Therefore $(dX_t)^2 = \sigma_t^2\,(dB_t)^2 = \sigma_t^2\,dt$.
Integration yields the result.
\end{proof}

\begin{remark}
The extra term $\frac{1}{2}f''(X_s)\,\sigma_s^2\,ds$, absent in
classical calculus, is called the \emph{It\^o correction term}. It
arises from the non-zero quadratic variation of Brownian motion.
\end{remark}

\begin{keyformulas}
\textbf{It\^o Calculus Rules}
\begin{align*}
  (dB_t)^2 &= dt, \qquad dt\cdot dB_t = 0, \qquad (dt)^2 = 0 \\[4pt]
  d(X_t Y_t) &= X_t\,dY_t + Y_t\,dX_t + dX_t\,dY_t
    \quad \text{(product rule)} \\[4pt]
  df(X_t) &= f'(X_t)\,dX_t + \tfrac{1}{2}f''(X_t)\,(dX_t)^2
\end{align*}
\end{keyformulas}

\begin{corollary}[Multidimensional It\^o's Formula]
Let $\mathbf{X}_t = (X_t^1, \ldots, X_t^d)$ be a vector of It\^o
processes and $f \in C^2(\R^d)$. Then:
\[
  df(\mathbf{X}_t) = \sum_{i=1}^d \frac{\partial f}{\partial x_i}\,dX_t^i
    + \frac{1}{2}\sum_{i,j=1}^d \frac{\partial^2 f}{\partial x_i\,\partial x_j}
    \,d\langle X^i, X^j\rangle_t.
\]
\end{corollary}

\begin{example}
Compute $d(B_t^2)$ where $B_t$ is a standard Brownian motion.
With $f(x) = x^2$, $f'(x) = 2x$, $f''(x) = 2$:
\[
  d(B_t^2) = 2B_t\,dB_t + dt.
\]
Thus $\int_0^t B_s\,dB_s = \frac{1}{2}(B_t^2 - t)$, showing that the
It\^o integral differs from the naive Riemann--Stieltjes result.
\end{example}

\section{Stochastic Differential Equations}
\label{sec:sde}
% ------------------------------------------------------------

\begin{definition}[Stochastic Differential Equation]
\index{stochastic differential equation}
An \textbf{SDE} is an equation of the form:
\[
  dX_t = b(t, X_t)\,dt + \sigma(t, X_t)\,dB_t, \qquad X_0 = x_0,
\]
where $b:[0,T]\times\R\to\R$ is the \emph{drift} coefficient and
$\sigma:[0,T]\times\R\to\R$ the \emph{diffusion} coefficient.
\end{definition}

\begin{theorem}[Existence and Uniqueness (It\^o--Lipschitz)]
\label{thm:existence_sde}
\index{existence theorem for SDEs}
If $b$ and $\sigma$ are Lipschitz in $x$ uniformly in $t$:
\[
  \abs{b(t,x) - b(t,y)} + \abs{\sigma(t,x) - \sigma(t,y)} \leq K\,\abs{x-y},
\]
and have linear growth: $\abs{b(t,x)} + \abs{\sigma(t,x)} \leq C(1+\abs{x})$,
then the SDE admits a unique strong solution $(X_t)_{0 \leq t \leq T}$
that is adapted, has continuous paths, and satisfies
$\E[\sup_{0 \leq t \leq T} X_t^2] < \infty$.
\end{theorem}

\begin{proof}[Proof sketch]
Use Picard iteration. Set $X_t^{(0)} = x_0$ and:
\[
  X_t^{(n+1)} = x_0 + \int_0^t b(s, X_s^{(n)})\,ds
    + \int_0^t \sigma(s, X_s^{(n)})\,dB_s.
\]
The Lipschitz condition and It\^o isometry give
$\E[\sup_{s \leq t} \abs{X_s^{(n+1)} - X_s^{(n)}}^2]
\leq (Ct)^n/n!$ (stochastic Gronwall lemma), yielding uniform $L^2$
convergence.
\end{proof}

\section{Geometric Brownian Motion}
\label{sec:gbm}
% ------------------------------------------------------------

\begin{definition}[Geometric Brownian Motion]
\index{geometric Brownian motion}
\textbf{Geometric Brownian motion} (GBM) is the solution of:
\[
  dS_t = \mu\,S_t\,dt + \sigma\,S_t\,dB_t, \qquad S_0 > 0.
\]
\end{definition}

\begin{theorem}[Explicit Solution of GBM]
\label{thm:gbm_solution}
The solution of GBM is:
\[
  S_t = S_0\exp\left[\left(\mu - \frac{\sigma^2}{2}\right)t
    + \sigma\,B_t\right].
\]
\end{theorem}

\begin{proof}
Set $Y_t = \ln S_t$. By It\^o's formula with $f(x) = \ln x$:
\[
  dY_t = \frac{1}{S_t}\,dS_t - \frac{1}{2S_t^2}\,(dS_t)^2
  = \mu\,dt + \sigma\,dB_t - \frac{\sigma^2}{2}\,dt
  = \left(\mu - \frac{\sigma^2}{2}\right)dt + \sigma\,dB_t.
\]
Integrating: $Y_t = Y_0 + (\mu - \sigma^2/2)\,t + \sigma B_t$.
\end{proof}

\begin{intuition}
GBM is the canonical model for financial asset prices (Black--Scholes
model). The term $-\sigma^2/2$ in the exponent is the It\^o correction:
without it, one would get $\E[S_t] = S_0 e^{(\mu+\sigma^2/2)t}$ instead
of the correct $\E[S_t] = S_0 e^{\mu t}$.
\end{intuition}

\section{The Ornstein--Uhlenbeck Process}
\label{sec:ou}
% ------------------------------------------------------------

\begin{definition}[Ornstein--Uhlenbeck Process]
\index{Ornstein--Uhlenbeck}
The \textbf{Ornstein--Uhlenbeck} process solves:
\[
  dX_t = -\theta\,(X_t - \mu)\,dt + \sigma\,dB_t, \qquad X_0 = x_0,
\]
where $\theta > 0$ is the mean-reversion rate, $\mu$ the long-run mean,
and $\sigma > 0$ the volatility.
\end{definition}

\begin{proposition}[Explicit Solution]
The solution is:
\[
  X_t = \mu + (x_0 - \mu)\,e^{-\theta t}
    + \sigma\int_0^t e^{-\theta(t-s)}\,dB_s.
\]
In particular, $X_t$ is Gaussian with $\E[X_t] = \mu + (x_0 - \mu)e^{-\theta t}$
and $\Var(X_t) = \frac{\sigma^2}{2\theta}(1 - e^{-2\theta t})$.
\end{proposition}

\begin{proof}
Set $Y_t = e^{\theta t}(X_t - \mu)$. By It\^o's formula,
$dY_t = \theta e^{\theta t}(X_t - \mu)\,dt + e^{\theta t}\,dX_t
= \sigma e^{\theta t}\,dB_t$.
Integrating: $Y_t = Y_0 + \sigma\int_0^t e^{\theta s}\,dB_s$,
so $X_t - \mu = (x_0 - \mu)e^{-\theta t} + \sigma\int_0^t
e^{-\theta(t-s)}\,dB_s$. The variance follows from the It\^o isometry.
\end{proof}

\begin{remark}
The Ornstein--Uhlenbeck process is the unique Gaussian process that is
simultaneously stationary (as $t \to \infty$), Markovian, and has
continuous paths. Its invariant measure is
$\mathcal{N}(\mu, \sigma^2/(2\theta))$.
\end{remark}

\section{The Langevin Equation}
\label{sec:langevin}
% ------------------------------------------------------------

\begin{definition}[Langevin Equation]
\index{Langevin equation}
The \textbf{Langevin equation} for a particle of mass $m$ subject to
friction $-\gamma v$ and thermal noise is:
\[
  m\,dv_t = -\gamma\,v_t\,dt + \sqrt{2\gamma k_B T}\,dB_t,
\]
where $k_B$ is Boltzmann's constant and $T$ the temperature.
\end{definition}

\begin{proposition}[Fluctuation-Dissipation Relation]
\index{fluctuation-dissipation}
At thermodynamic equilibrium, the velocity variance satisfies:
\[
  \Var(v_\infty) = \frac{k_B T}{m},
\]
consistent with the equipartition theorem. This relation links the
noise intensity ($2\gamma k_B T$) to the friction coefficient ($\gamma$).
\end{proposition}

\begin{exercise}
Show that $\int_0^t B_s\,dB_s = \frac{1}{2}(B_t^2 - t)$ using It\^o's
formula. Interpret the difference with the deterministic result
$\int_0^t x\,dx = x^2/2$.
\end{exercise}

\begin{exercise}
Solve the SDE $dX_t = X_t\,dt + X_t\,dB_t$ with $X_0 = 1$.
Compute $\E[X_t]$ and $\Var(X_t)$.
\end{exercise}

\begin{exercise}
Simulate the Ornstein--Uhlenbeck process numerically using the
Euler--Maruyama scheme\index{Euler--Maruyama}:
\[
  X_{n+1} = X_n - \theta(X_n - \mu)\,\Delta t
    + \sigma\sqrt{\Delta t}\,Z_n, \quad Z_n \sim \mathcal{N}(0,1).
\]
Verify that the stationary distribution is
$\mathcal{N}(\mu, \sigma^2/(2\theta))$.
\end{exercise}

\begin{exercise}
Let $f(t,x) = e^{-rt}\,x^\alpha$ and $X_t$ be a GBM with parameters
$(\mu, \sigma)$. Apply It\^o's formula to compute $df(t,X_t)$ and
determine the value of $\alpha$ for which $f(t,X_t)$ is a martingale.
\end{exercise}

\begin{exercise}[Stratonovich--It\^o conversion]
Consider the Stratonovich SDE:
\[
  dX_t = X_t \circ dB_t.
\]
\begin{enumerate}
  \item Recall the general relationship between the Stratonovich integral
        $\int_0^t \sigma(X_s) \circ dB_s$ and the It\^o integral, involving
        the correction term $\frac{1}{2}\int_0^t \sigma'(X_s)\,\sigma(X_s)\,ds$.
  \item Convert the above SDE into an It\^o SDE.
  \item Solve the resulting It\^o SDE and verify that the solution is
        $X_t = X_0\,e^{B_t}$. Compare with GBM having parameters
        $\mu = \sigma^2/2$, $\sigma = 1$.
\end{enumerate}
\end{exercise}

\begin{exercise}[Girsanov's theorem]
Let $(B_t)_{0 \leq t \leq T}$ be a standard Brownian motion under $\PP$
and let $\theta \in \R$ be a constant. Define the exponential martingale:
\[
  Z_t = \exp\!\left(\theta B_t - \frac{\theta^2}{2}\,t\right).
\]
\begin{enumerate}
  \item Verify using It\^o's formula that $Z_t$ is a martingale under $\PP$.
  \item Define the measure $\mathbb{Q}$ by $d\mathbb{Q} = Z_T\,d\PP$.
        Show that $\widetilde{B}_t = B_t - \theta t$ is a standard
        Brownian motion under $\mathbb{Q}$ (Girsanov's theorem).
  \item Application: let $X_t$ solve
        $dX_t = \mu\,dt + \sigma\,dB_t$ under $\PP$. Find a measure
        $\mathbb{Q}$ under which $X_t$ is a (driftless) Brownian motion.
        Give $d\mathbb{Q}/d\PP$ explicitly.
\end{enumerate}
\end{exercise}

\begin{exercise}[Strong vs.\ weak solutions]
Consider Tanaka's SDE:
\[
  dX_t = \sign(X_t)\,dB_t, \qquad X_0 = 0,
\]
where $\sign(x) = \ind_{x > 0} - \ind_{x \leq 0}$.
\begin{enumerate}
  \item Show that the diffusion coefficient $\sigma(x) = \sign(x)$ fails
        to be Lipschitz at $x = 0$. Why does the It\^o--Lipschitz strong
        existence theorem not apply?
  \item Show that if $X_t$ is a solution, then $Y_t = \abs{X_t}$
        satisfies $Y_t = \abs{B_t}$ in law (you may use Tanaka's formula).
        Deduce that a \emph{weak} solution exists.
  \item Explain why no \emph{strong} solution exists (i.e., no solution
        measurable with respect to the filtration generated by $B$).
        \emph{Hint}: show that $\sign(X_t)$ cannot be determined from the
        Brownian motion $B$ alone.
\end{enumerate}
\end{exercise}
