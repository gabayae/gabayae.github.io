% ============================================================
\chapter{Martingales --- Continuous Time}
\label{ch:continuous_martingales}
% ============================================================

The passage from discrete-time to continuous-time martingales is one of the most delicate transitions in stochastic process theory. In discrete time, one conditions on a finite past; in continuous time, the filtration $(\mathcal{F}_t)_{t \geq 0}$ forms a continuum of information, and subtle phenomena emerge: path regularity, uncountable stopping times, maximal inequalities. Joseph Doob, in the 1950s, showed that every continuous-time martingale admits a c\`adl\`ag modification (right-continuous with left limits), a technical but essential result. Doob's inequalities, the optional stopping theorem, and the Doob--Meyer decomposition form the foundation on which all of stochastic calculus rests --- and, by extension, modern mathematical finance.

\section{Definitions}

\begin{definition}[Continuous-Time Martingale]
Let $(\Omega, \mathcal{F}, (\mathcal{F}_t)_{t \geq 0}, \PP)$ be a filtered
probability space satisfying the usual conditions. An adapted process
$(M_t)_{t \geq 0}$ is a \emph{martingale} (in continuous time) if:
\begin{enumerate}
  \item $\E[\abs{M_t}] < \infty$ for all $t \geq 0$,
  \item $\E[M_t \mid \mathcal{F}_s] = M_s$ a.s.\ for all $0 \leq s \leq t$.
\end{enumerate}
\end{definition}

\begin{definition}[Local Martingale]
An adapted continuous process $(M_t)_{t \geq 0}$ is a \emph{local martingale}
if there exists an increasing sequence of stopping times $(\tau_n)_{n \geq 1}$
with $\tau_n \uparrow \infty$ a.s.\ such that for each $n$, the stopped process
$(M_{t \wedge \tau_n})_{t \geq 0}$ is a martingale.

The sequence $(\tau_n)$ is called a \emph{localising sequence}.
\end{definition}

\begin{warning}[title=\textbf{Local Martingale $\neq$ Martingale}]
Every martingale is a local martingale, but the converse is false. A positive
local martingale is a supermartingale (by Fatou's lemma), but not necessarily a
martingale.
\end{warning}

\section{Path Regularity}

\begin{theorem}[C\`adl\`ag Modification]
\label{thm:cadlag}
Let $(M_t)_{t \geq 0}$ be a real-valued martingale (or supermartingale). If
$t \mapsto \E[M_t]$ is right-continuous, then $(M_t)$ admits a c\`adl\`ag
modification.
\end{theorem}

\begin{remark}
Under the usual conditions on the filtration, the c\`adl\`ag modification is
still a martingale. Henceforth, we assume continuous-time martingales are
c\`adl\`ag.
\end{remark}

\begin{theorem}[Continuity of $L^2$ Martingales]
If $(M_t)$ is a square-integrable martingale with continuous sample paths, then
the quadratic variation process $\langle M \rangle_t$ is continuous and
$(M_t^2 - \langle M \rangle_t)$ is a martingale.
\end{theorem}

\section{Doob's Inequalities in Continuous Time}

\begin{theorem}[Doob's Maximal Inequality]
Let $(M_t)_{0 \leq t \leq T}$ be a c\`adl\`ag martingale. For all $\lambda > 0$:
\[
  \PP\Bigl(\sup_{0 \leq t \leq T} \abs{M_t} \geq \lambda\Bigr)
  \leq \frac{\E[\abs{M_T}]}{\lambda}.
\]
\end{theorem}

\begin{theorem}[Doob's $L^p$ Inequality --- Continuous Time]
For $p > 1$ and a c\`adl\`ag martingale $(M_t)$:
\[
  \E\Bigl[\sup_{0 \leq t \leq T} \abs{M_t}^p\Bigr]
  \leq \Bigl(\frac{p}{p-1}\Bigr)^p \E[\abs{M_T}^p].
\]
\end{theorem}

\begin{theorem}[Burkholder--Davis--Gundy Inequality]
\label{thm:BDG}
There exist universal constants $c_p, C_p > 0$ such that for every continuous
local martingale $(M_t)$ with $M_0 = 0$ and for all $p > 0$:
\[
  c_p \, \E[\langle M \rangle_T^{p/2}]
  \leq \E\Bigl[\sup_{0 \leq t \leq T} \abs{M_t}^p\Bigr]
  \leq C_p \, \E[\langle M \rangle_T^{p/2}].
\]
For $p = 2$, one may take $c_2 = 1$ and $C_2 = 4$ (Doob's inequality).
\end{theorem}

\section{Quadratic Variation}

\begin{definition}[Quadratic Variation]
\label{def:quadratic_variation}
Let $(M_t)_{t \geq 0}$ be a continuous local martingale. The \emph{quadratic
variation} (or \emph{bracket}) of $M$ is the unique continuous increasing
process $(\langle M \rangle_t)_{t \geq 0}$ with $\langle M \rangle_0 = 0$ such
that $(M_t^2 - \langle M \rangle_t)$ is a local martingale.

It has the representation:
\[
  \langle M \rangle_t = \lim_{\abs{\Pi} \to 0}
  \sum_{k=0}^{n-1} (M_{t_{k+1}} - M_{t_k})^2 \quad \text{(limit in probability)},
\]
where $\Pi = \{0 = t_0 < t_1 < \cdots < t_n = t\}$ is a partition of $[0, t]$.
\end{definition}

\begin{definition}[Cross Variation]
For two continuous local martingales $M$ and $N$, the \emph{cross variation}
(or \emph{cross bracket}) is
\[
  \langle M, N \rangle_t = \frac{1}{4}\bigl(\langle M + N \rangle_t
  - \langle M - N \rangle_t\bigr),
\]
and is the unique continuous finite-variation process such that
$(M_t N_t - \langle M, N \rangle_t)$ is a local martingale.
\end{definition}

\begin{keyformulas}[title=\textbf{Fundamental Quadratic Variations}]
For Brownian motion $(B_t)$:
\begin{align*}
  \langle B \rangle_t &= t \\
  \langle B^{(i)}, B^{(j)} \rangle_t &= \delta_{ij} t
    \quad \text{(multidimensional Brownian motion)}
\end{align*}
\end{keyformulas}

\section{Doob--Meyer Decomposition}

\begin{theorem}[Doob--Meyer Decomposition]
\label{thm:doob_meyer}
Let $(X_t)_{t \geq 0}$ be a c\`adl\`ag supermartingale of class (D). Then there
exists a unique decomposition:
\[
  X_t = M_t - A_t,
\]
where $(M_t)$ is a c\`adl\`ag martingale and $(A_t)$ is an increasing predictable
process with $A_0 = 0$.
\end{theorem}

\begin{definition}[Class (D)]
A supermartingale $(X_t)$ is of \emph{class (D)} if the family
$\{X_{\tau} : \tau \text{ finite stopping time}\}$ is uniformly integrable.
\end{definition}

\begin{remark}
The Doob--Meyer decomposition is the continuous-time analogue of the discrete
Doob decomposition. It is fundamental for constructing the stochastic integral.
\end{remark}

\section{Optional Stopping in Continuous Time}

\begin{theorem}[Optional Stopping --- Continuous Time]
\label{thm:ost_continuous}
Let $(M_t)_{t \geq 0}$ be a continuous uniformly integrable martingale and
$\tau$ a stopping time (possibly infinite). Then
\[
  \E[M_{\tau}] = \E[M_0].
\]
More generally, if $\sigma \leq \tau$ are stopping times:
\[
  \E[M_{\tau} \mid \mathcal{F}_{\sigma}] = M_{\sigma} \quad \text{a.s.}
\]
\end{theorem}

\begin{theorem}[Stopping for Positive Supermartingales]
\label{thm:stop_supermartingale}
If $(M_t)_{t \geq 0}$ is a positive continuous supermartingale and $\tau$ is a
stopping time, then
\[
  \E[M_{\tau}] \leq \E[M_0].
\]
\end{theorem}

\section{Martingale Representation}

\begin{theorem}[Brownian Martingale Representation]
\label{thm:representation}
Let $(B_t)_{t \geq 0}$ be a Brownian motion and $(\mathcal{F}_t)$ its completed
natural filtration. Every continuous local martingale $(M_t)$ with respect to
$(\mathcal{F}_t)$ admits the representation:
\[
  M_t = M_0 + \int_0^t H_s \, dB_s,
\]
where $(H_t)$ is a predictable process with $\int_0^t H_s^2 \, ds < \infty$
a.s.\ for all $t$.
\end{theorem}

\begin{intuition}[title=\textbf{Completeness of the Brownian Filtration}]
This theorem says that the Brownian filtration is ``complete'' in the sense
that every martingale can be written as a stochastic integral with respect to
Brownian motion. This is a central result for applications in finance (market
completeness theorem).
\end{intuition}

\section{Girsanov's Theorem}

\begin{theorem}[Girsanov]
\label{thm:girsanov}
Let $(B_t)_{t \geq 0}$ be a Brownian motion under $\PP$ and $(\theta_t)_{t \geq 0}$
an adapted process with $\int_0^T \theta_s^2 \, ds < \infty$ a.s. Define the
Dol\'eans--Dade exponential:
\[
  Z_t = \exp\Bigl(-\int_0^t \theta_s \, dB_s - \frac{1}{2} \int_0^t \theta_s^2 \, ds\Bigr).
\]
If $(Z_t)$ is a martingale (Novikov's condition:
$\E[\exp(\frac{1}{2}\int_0^T \theta_s^2 \, ds)] < \infty$), then under the
measure $\mathbb{Q}$ defined by
$d\mathbb{Q}/d\PP\big|_{\mathcal{F}_T} = Z_T$, the process
\[
  \tilde{B}_t = B_t + \int_0^t \theta_s \, ds
\]
is a Brownian motion.
\end{theorem}

\section{Convergence in Continuous Time}

\begin{theorem}[Continuous Martingale Convergence]
\label{thm:continuous_convergence}
Let $(M_t)_{t \geq 0}$ be a right-continuous supermartingale with
$\sup_{t \geq 0} \E[M_t^+] < \infty$. Then $M_{\infty} = \lim_{t \to \infty} M_t$
exists a.s.\ and $M_{\infty} \in L^1$.
\end{theorem}

\section{Exercises}

\begin{exercise}
Let $(B_t)_{t \geq 0}$ be a Brownian motion. Verify that
$M_t = \exp(\theta B_t - \theta^2 t/2)$ is a martingale for all $\theta \in \R$.
\end{exercise}

\begin{exercise}
Prove that if $(M_t)$ is a continuous positive local martingale, then it is a
supermartingale. \emph{Hint: use Fatou's lemma.}
\end{exercise}

\begin{exercise}
Let $(M_t)$ be a continuous square-integrable martingale. Show that
$\E[M_t^2] = \E[M_0^2] + \E[\langle M \rangle_t]$.
\end{exercise}

\begin{exercise}
Compute $\langle M \rangle_t$ for $M_t = B_t^2 - t$ where $(B_t)$ is a
Brownian motion.
\end{exercise}

\begin{exercise}
Let $(B_t)$ be a Brownian motion and $\tau_a = \inf\{t \geq 0 : B_t = a\}$.
Use optional stopping with the exponential martingale to compute
$\E[e^{-\alpha \tau_a}]$ for $\alpha > 0$ and $a > 0$.
\end{exercise}

\begin{exercise}
State and prove Girsanov's theorem in the case where $\theta_t = \theta$ is
constant. Verify Novikov's condition.
\end{exercise}

\begin{exercise}
Show that the Burkholder--Davis--Gundy inequality (Theorem~\ref{thm:BDG})
implies Doob's $L^2$ inequality for continuous martingales.
\end{exercise}
